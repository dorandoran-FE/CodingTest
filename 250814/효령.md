안전영역

```python3

# 어떤 지역의 높이 정보를 파악한다. 그 다음에 잠기지 않는 안전 영역이 최대로 몇 개가 들어가는지 조사
# 비의 양에 따라 일정한 높이 이하의 모든 지점은 물에 잠김
# 어떤지역의 높이 정보(H) = [...][...]
# 모든 강수량에 대한 안전영역 계산 -> 최댓값 출력

import sys
input = sys.stdin.readline

# 입력값
N = int(input())
graph = [list(map(int, input().split())) for _ in range(N)]

# 상하좌우 방향
dx = [-1, 1, 0 , 0]
dy = [0, 0, -1 , 1]

# dfs 함수
def dfs(x, y, h, visited):
  visited[x][y] = True
  for i in range(4):
    # 그다음 옆집 이동
    nx, ny = x + dx[i] , y + dy[i]
    # 범위안, 방문 안함, 물에 안잠김 체크
    if 0 <= nx < N and 0 <= ny < N:
      if not visited[nx][ny] and graph[nx][ny] > h:
        dfs(nx, ny , h, visited)

# 최대 높이값
max_height = max(map(max , graph)) # 각 리스트에서 최댓값 뽑아줌
result = 0

# 비 높이 h를 0 ~ 최댓값 까지 탐방해서 result 뽑기
for h in range(max_height + 1):
  # 새로 스티커 지도 만들기
  visited = [[False] * N for _ in range(N)]
  # 안전동네 갯수
  safe_count = 0

  for i in range(N):
    for j in range(N):
      # 안전동네 + 아직 방문 안했을 때 dfs 시작
      if graph[i][j] > h and not visited[i][j]:
        dfs(i, j, h, visited)
        safe_count += 1

  result = max(result, safe_count)

print(result)

```
