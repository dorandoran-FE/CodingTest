```python3
# 수 찾기, 첫째 줄에 자연수 ~~ 다음줄에 n개의 정수
# 다음 줄에는 m개의 수들이 주어지는 a 안에 존재하는지 알아내면 돈다 .. 근데 a는 2번째줄의 배열인 것 같음
#세번째 줄에는 m개의 수들이 주어진다. 이 수들이 a안에 존재하는지 1, 0으로 return 을 하면 된다


import sys 
input = sys.stdin.readline

# 입력값 받기
N = int(input())
A = list(map(int, input().split()))
M = int(input())
queries = list(map(int, input().split()))

# set 집합에 A 다 넣어놓고 체크하기
S = set(A)  

for x in queries:
    if x in S:   # S 집합에 있으면 1
        print(1)
    else:        # 없으면
        print(0)
        
```
