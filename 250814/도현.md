### 입국 심사

``` javascript
function solution(n, times) {    
    let low = 1
    let high = Math.max(...times) * n
    let answer = high;\
    
    while(low <= high) {
        const mid = Math.floor((low + high) / 2)
        const people = times.reduce((acc, time) => acc + Math.floor(mid / time), 0)
        
        if (people >= n) {
            answer = mid
            high = mid - 1
        } else {
            low = mid + 1
        }
    }
    
    return answer;
}
```


### 수 찾기

``` javascript
const filePath = process.platform === 'linux' ? 'dev/stdin' : './input.txt';
const input = require('fs').readFileSync(filePath).toString().trim().split('\n');
const A = input[1].split(' ').map(v=>+v);
const B = input[3].split(' ').map(v=>+v);

A.sort((a,b) => a - b)

const result = []

B.forEach(m => {
    let start = 0
    let end = A.length - 1
    let isIn = false
    
    while(start <= end) {
        let mid = Math.floor((start + end) / 2)
        if (m < A[mid]) {
            end = mid - 1
        }
        else if (m > A[mid]) {
            start = mid + 1
        }
        else {
            isIn = true
            break
        }
    }
    if (isIn) result.push(1)
    else result.push(0)
})

console.log(result.join('\n'))
```
