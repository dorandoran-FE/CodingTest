```python

import sys

# 어떤 지역의 높이 정보가 주어졌을 때, 물에 잠기지 않는 안전한 영역의 최대 개수를 계산
# 어떤 지역 => 2차원 배열의 행과 열의 개수를 나타내는 수 N이 입력된다.

# 비의 높이를 0부터 최대높이까지 반복하면서 도출할 수 있을 것 같음
# 비에 잠기지 않은 곳만 탐색 (높이 > 비 높이)

# 1. 입력 받기
n = int(input())
graph = [list(map(int, input().split())) for _ in range(n)]

# 2. 상하좌우 이동
dx = [-1, 1, 0, 0]  # 위, 아래
dy = [0, 0, -1, 1]  # 왼쪽, 오른쪽

# 3. dfs 함수 정의
def dfs(x, y, height, visited):
    # 현재 위치 방문 처리
    visited[x][y] = True

    # 상하좌우 방향 탐색
    for i in range(4):
        nx = x + dx[i]
        ny = y + dy[i]

        # 범위를 벗어나지 않고, 방문하지 않았고, 물에 잠기지 않은 곳이면
        if 0 <= nx < n and 0 <= ny < n:
            if not visited[nx][ny] and graph[nx][ny] > height:
                dfs(nx, ny, height, visited)

# 4. 가장 높은 높이 찾는 법
max_height = max(map(max, graph))  # 그래프 내에서 가장 높은 숫자 찾기

# 5. 결과 도출 (비의 높이 0 ~ 최대 높이까지 탐색하기)
result = 0  

for h in range(0, max_height + 1):  # 비의 높이를 0부터 max까지 모두 탐색
    visited = [[False] * n for _ in range(n)]  # 방문 체크 배열 초기화
    count = 0  # 안전한 영역 갯수

    for i in range(n):
        for j in range(n):
            # 아직 방문 안했고, 물에 잠기지 않은 지역
            if not visited[i][j] and graph[i][j] > h:
                dfs(i, j, h, visited)  # 해당 지점부터 연결된 영역 탐색
                count += 1  # 안전 영역 개수 증가

    # 최대값 갱신
    result = max(result, count)

# 6. 결과 출력
print(result)

```
