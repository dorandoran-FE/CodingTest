def solution(array, commands):
    # array : 배열
    # commands : 2차원 배열 [array][i,j,k]
    # 2차원 배열을 어떻게 활용해야될지 모르겠음 ,, 
    # array에서 commands 에 0,1,2 로 있는 것들을 자르는거니까..
    
    answer = []

    for command in commands:
        
        i = command[0]
        j = command[1]
        k = command[2]
    
        slice_array = array[i-1:j]
        slice_array.sort()
    
        n = slice_array[k-1]
    
    return n
