	int[] dx = {1, 0, -1, 0};

	public int solution(int[][] maps) {

		int maxY = maps.length;

        map[0][0] = 1;

		while (!queue.isEmpty()) {

			int currentY = currentQueue[1];

				int nextY = currentY + dy[i];

			}

	}
}
