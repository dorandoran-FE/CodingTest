### 단지 번호 붙이기

``` python
from collections import deque

N = int(input())
arr = []
ans = []

for i in range(N):
    temp = list(map(int, input()))
    arr.append(temp)

n, m = len(arr), len(arr[0])

dx = [-1,1,0,0]
dy = [0,0,-1,1]         

def bfs(a, b):
    queue = deque()
    queue.append((a, b))
    arr[a][b] = 0
    count = 1

    while queue:
        x, y = queue.popleft()
        for i in range(4):
            nx = dx[i] + x
            ny = dy[i] + y

            if nx < 0 or nx >= n or ny < 0 or ny >= m : 
                continue
                
            if arr[nx][ny] == 0 :
                continue
            
            if arr[nx][ny] == 1:
                queue.append((nx, ny))
                arr[nx][ny] = 0
                count += 1
    return count

for i in range(N):
    for j in range(N) :
        if arr[i][j] == 1:
            ans.append(bfs(i, j))
ans.sort()
print(len(ans))

for i in range(len(ans)):
    print(ans[i])
```



### 안전영역

```python
from collections import deque

N = int(input())
arr = []
max_num = 0
for i in range(N):
    temp = list(map(int, input().split()))
    for j in temp :
        if j >= max_num:
            max_num = j
    arr.append(temp)

n, m = len(arr), len(arr[0])

dx = [-1,1,0,0]
dy = [0,0,-1,1]

def bfs(a,b, h, visited):
    queue = deque()
    queue.append((a,b))
    visited[a][b] = 1

    while queue : 
        x, y = queue.popleft()

        for i in range(4):
            nx = x + dx[i]
            ny = y + dy[i]

            if nx < 0 or nx >= n or ny < 0 or ny >= m:
                continue

            #비의 높이를 어떻게 컨트롤?
            if arr[nx][ny] > h and visited[nx][ny] == 0 :
                visited[nx][ny] = 1
                queue.append((nx, ny))


ans = 0
for t in range(max_num):
    cnt = 0
    visited = [[0] * N for i in range(N)]
    for i in range(N) :
        for j in range(N) :
            if arr[i][j] > t and visited[i][j] == 0:
                bfs(i, j, t, visited)
                cnt += 1
    if ans < cnt : 
        ans = cnt
print(ans)

```
