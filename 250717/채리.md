``` javascript
function solution(n, lost, reserve) {
  let answer = 0;
  const student = Array(n + 2).fill(0); // [0] * (n+2)

  for (let i of reserve) {
    student[i] += 1;
  }

  for (let j of lost) {
    student[j] -= 1;
  }

  // console.log(student);

  for (let i = 1; i <= n; i++) {
    if (student[i] > 0) {
      // 앞 학생이 체육복이 없을 때
      if (student[i - 1] < 0) {
        student[i] -= 1;
        student[i - 1] += 1;
      }
      // 뒷 학생이 체육복이 없을 때
      else if (student[i + 1] < 0 && student[i] > 0) {
        student[i] -= 1;
        student[i + 1] += 1;
      }
    }
  }

  for (let i = 1; i <= n; i++) {
    if (student[i] >= 0) {
      answer += 1;
    }
  }

  return answer;
}
```
