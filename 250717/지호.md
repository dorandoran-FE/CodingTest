### 체육복

```java
public class Solution {
	public int solution(int n, int[] lost, int[] reserve) {
		int[] students = new int[n + 2]; // +2 to handle edge cases easily
		// 1씩 넣어주자
		for (int i = 1; i <= n; i++) {
			students[i] = 1; // 모든 학생은 기본적으로 체육복을 가지고 있다.
		}

		// 잃어버린 학생들 처리
		for (int l : lost) {
			students[l]--; // 잃어버린 학생은 체육복이 하나 줄어든다.
		}

		// 여분의 학생
		for (int r : reserve) {
			students[r]++; // 여분 학생
		}

		// 비교
		for (int i = 1; i <= n; i++) {
			if (students[i] == 0) {
				// 체육복 없고 앞에 학생에게
				if (students[i - 1] > 1) {
					students[i - 1]--;
					students[i]++;
				} else if (students[i + 1] > 1) {
					students[i + 1]--;
					students[i]++;
				}
			}
		}

		// 1이상인 것의 개수
		return (int)Arrays.stream(students).filter(value -> value >= 1).count();
	}
}
```
